function busqueda_binaria(arr, target, left, right)
begin
    if left > right
    begin
        return 0
    end
    else
    begin
        middle = (left + right) / 2
        if arr[middle] == target
        begin
            return middle
        end
        else
        begin
            if arr[middle] > target
            begin
                return call busqueda_binaria(arr, target, left, middle - 1)
            end
            else
            begin
                return call busqueda_binaria(arr, target, middle + 1, right)
            end
        end
    end
end
